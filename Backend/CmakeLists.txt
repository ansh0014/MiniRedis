cmake_minimum_required(VERSION 3.10)
project(miniredis_backend CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set vcpkg toolchain
set(CMAKE_TOOLCHAIN_FILE "D:/cpp-tools/vcpkg/scripts/buildsystems/vcpkg.cmake" CACHE STRING "Vcpkg toolchain file")

# Find required packages
find_package(Drogon CONFIG REQUIRED)
find_package(Boost REQUIRED COMPONENTS uuid)

# Find redis++ (optional, but recommended)
find_package(redis++ CONFIG)
if(redis++_FOUND)
    message(STATUS "Found redis++")
    set(REDIS_LIBS redis++::redis++)
else()
    message(WARNING "redis++ not found, some features disabled")
    set(REDIS_LIBS "")
endif()

# Source files
set(SOURCES
    main.cc
    controller/ApiController.cc
)

# Create executable
add_executable(${PROJECT_NAME} ${SOURCES})

# Include directories
target_include_directories(${PROJECT_NAME} PRIVATE 
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/controller
    ${Boost_INCLUDE_DIRS}
)

# Link libraries
target_link_libraries(${PROJECT_NAME} PRIVATE 
    Drogon::Drogon
    Boost::uuid
    ${REDIS_LIBS}
)

# Windows-specific libraries
if(WIN32)
    target_link_libraries(${PROJECT_NAME} PRIVATE ws2_32)
endif()
